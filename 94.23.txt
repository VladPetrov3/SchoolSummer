int main()
{
    //94
    vector<int> f(1000, 0);
    f[22] = 1;
    for (int i = 22; i >= 2; i--) {
        if (i - 1 >= 2) f[i - 1] += f[i];
        if (i - 3 >= 2) f[i - 3] += f[i];
        if (i >= 4) {
            if (i % 4 >= 2) {
                f[i % 4] += f[i];
            }
        }
    }
    cout << f[2];
}
